def insertar_back(valor)
  n = Node.new(valor,nil,@tail)

def to_s
  if empty
    return "H -> nil <- T"
  end
  cad = "H -> "
  current = @tail
  until current == nil
    cad = cad + current.value.to_s
    current = current.prev
    if (current != nil)
      cad = cad + " <-> "
    end
  end
  cad = cad + " <- T"
end
