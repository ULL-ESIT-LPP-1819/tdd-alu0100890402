def imc_descripcion
  case imc_value
  when (1)...(18.5)
    "Bajo peso"
  when (18.5)..(24.9)
    "Adecuado"
  when (25)..(29.9)
    "Sobrepeso"
  else
    "Obesidad"
  end
end

def rcc_riesgo
  if(@sexo == 1)
    case rcc_value
    when 0..(0.88)
      "Bajo"
    when (0.88)..(0.95)
      "Moderado"
    when (0.95)..(1.01)
      "Alto"
    else
      "Muy alto"
    end
  else
    case rcc_value
    when 0..(0.75)
      "Bajo"
    when (0.75)..(0.82)
      "Moderado"
    else
      "Alto"
    end
  end
end
